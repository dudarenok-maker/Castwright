# Use the maintained Coqui fork (the original `TTS` package was discontinued
# in early 2024 and won't install on Python >=3.12). `coqui-tts` keeps the
# `from TTS.api import TTS` import surface and supports Python 3.10–3.13.
# Pinned soft-floor to a recent stable; let pip pick the best 3.x compatible
# wheel. The `[codec]` extra pulls torchcodec, which PyTorch 2.9+ requires
# for audio I/O (the symbols `torchaudio.load`/`save` moved out of core).
coqui-tts[codec]>=0.24.0
# coqui-tts internals import `isin_mps_friendly` from transformers.pytorch_utils,
# which is a 4.x-line private util. transformers 5.x removed/renamed it, so
# pip happily resolves to a too-new wheel and the import explodes at runtime.
# Pin to the 4.x line until coqui-tts updates its compat layer.
transformers>=4.45,<5.0
fastapi>=0.115,<0.116
uvicorn[standard]>=0.30,<0.32
# Numpy 2.x is required by kokoro-onnx>=0.4.0; coqui-tts 0.27.x only floors
# at numpy>=1.26 (no upper cap) so the bump is safe. The earlier `<2.0`
# pin was defensive (predates coqui-tts updating its compat layer); lifting
# it is the price of admission for installing Kokoro alongside.
numpy>=1.26,<3.0
# NOTE on DeepSpeed: not in requirements because its setup.py needs the
# standalone NVIDIA CUDA Toolkit (with nvcc) at install time — the CUDA
# runtime libraries that ship inside the PyTorch wheel aren't enough. On
# Windows boxes without the toolkit, `pip install deepspeed` fails at
# metadata-generation. If you have (or are willing to install) the CUDA
# Toolkit from https://developer.nvidia.com/cuda-downloads, then:
#   DS_BUILD_OPS=0 ./.venv/Scripts/python.exe -m pip install --no-build-isolation "deepspeed>=0.15,<0.16"
# and set COQUI_DEEPSPEED=1 in server/.env. The sidecar already handles
# both states — it logs a warning and continues in vanilla mode if
# DeepSpeed isn't available.

# Kokoro v1 — quality-tuned local TTS engine, eagerly loaded at sidecar startup.
# The [gpu] extra pulls onnxruntime-gpu ONLY on x86_64 non-macOS (kokoro-onnx's
# own platform marker); plain onnxruntime always comes from the core dep, so a
# macOS / Apple Silicon install gets the CPU/CoreML runtime and `pip install`
# succeeds. Run scripts/install-kokoro.{ps1,sh} once to fetch the model + voices
# manifest into voices/kokoro/ (gitignored, ~330 MB).
kokoro-onnx[gpu]>=0.4.0,<0.5.0

# Qwen3-TTS — per-character bespoke voices via design -> clone -> cache ->
# reuse (plan 108). NOT a fixed-voice catalog like Kokoro; voices are designed
# from a persona once and reused for consistency. Pulls torch + soundfile.
# Left unpinned here because the weights + exact wheel are fetched by
# scripts/install-qwen3.mjs (a few GB, gitignored under voices/qwen/), not on a
# plain `pip install -r`. NOTE: qwen-tts may pull a transformers newer than the
# `<5.0` cap coqui-tts needs above — if pip reports a conflict at install time,
# run the sidecar with EITHER Coqui OR Qwen in its venv, or pin transformers to
# a version both accept (verified empirically when the model is downloaded).
# qwen-tts

# faster-whisper — ASR (speech-to-text) for the per-sentence content-QA gate
# (srv-31): transcribe a synthesised sentence and word-error-rate it against the
# manuscript text to catch "fluent but wrong words" generations the signal-based
# segment QA can't see. CTranslate2 backend — small wheel, int8, CPU-fast and
# GPU-optional. CPU-first by default (ASR_DEVICE=cpu → zero VRAM); ASR_DEVICE=cuda
# opts into the GPU with a tiny/base int8 model (~150–400 MB). The model weights
# (~145 MB for `base`) are fetched by faster-whisper on first load from its own
# cache, not on a plain `pip install -r`. Off by default at the server (SEG_ASR_
# ENABLED=0), so this dep is dormant until the gate is switched on.
faster-whisper>=1.0,<2.0

# psutil — process RSS / committed-private bytes for the host-memory watchdog +
# /debug/memory readout (the 2026-05-30 host-RAM leak guard). Already arrives
# transitively via accelerate, but the watchdog now depends on it directly, so
# pin it explicitly rather than rely on a transitive that a future install could
# drop. Tiny, pure-bookkeeping dep; no native build.
psutil>=5.9
