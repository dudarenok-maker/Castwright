# Use the maintained Coqui fork (the original `TTS` package was discontinued
# in early 2024 and won't install on Python >=3.12). `coqui-tts` keeps the
# `from TTS.api import TTS` import surface and supports Python 3.10–3.13.
# Pinned soft-floor to a recent stable; let pip pick the best 3.x compatible
# wheel. The `[codec]` extra pulls torchcodec, which PyTorch 2.9+ requires
# for audio I/O (the symbols `torchaudio.load`/`save` moved out of core).
coqui-tts[codec]>=0.24.0
# coqui-tts internals import `isin_mps_friendly` from transformers.pytorch_utils,
# which is a 4.x-line private util. transformers 5.x removed/renamed it, so
# pip happily resolves to a too-new wheel and the import explodes at runtime.
# Pin to the 4.x line until coqui-tts updates its compat layer.
transformers>=4.45,<5.0
fastapi>=0.115,<0.116
uvicorn[standard]>=0.30,<0.32
# Numpy 2.x is required by kokoro-onnx>=0.4.0; coqui-tts 0.27.x only floors
# at numpy>=1.26 (no upper cap) so the bump is safe. The earlier `<2.0`
# pin was defensive (predates coqui-tts updating its compat layer); lifting
# it is the price of admission for installing Kokoro alongside.
numpy>=1.26,<3.0
# NOTE on DeepSpeed: not in requirements because its setup.py needs the
# standalone NVIDIA CUDA Toolkit (with nvcc) at install time — the CUDA
# runtime libraries that ship inside the PyTorch wheel aren't enough. On
# Windows boxes without the toolkit, `pip install deepspeed` fails at
# metadata-generation. If you have (or are willing to install) the CUDA
# Toolkit from https://developer.nvidia.com/cuda-downloads, then:
#   DS_BUILD_OPS=0 ./.venv/Scripts/python.exe -m pip install --no-build-isolation "deepspeed>=0.15,<0.16"
# and set COQUI_DEEPSPEED=1 in server/.env. The sidecar already handles
# both states — it logs a warning and continues in vanilla mode if
# DeepSpeed isn't available.

# Kokoro v1 — quality-tuned local TTS engine, eagerly loaded at sidecar
# startup. The ONNX runtime auto-selects CUDA when onnxruntime-gpu is
# installed and falls back to CPU otherwise. Tight pins because OneDrive-
# under-sync makes wheel re-resolution expensive (WinError 5 on partial
# installs). Run scripts/install-kokoro.ps1 once to download the model +
# voices manifest into voices/kokoro/ (gitignored, ~330 MB).
kokoro-onnx>=0.4.0,<0.5.0
onnxruntime-gpu>=1.20.0,<2.0.0
