# ships cores for FFmpeg 4–7 (we ship FFmpeg 8). We pin torch 2.8 (< 2.9), so
# torchaudio keeps in-core audio I/O and torchcodec is unnecessary.
coqui-tts>=0.24.0

# torch + torchaudio — EXPLICIT, matched, pinned (coqui-tts 0.27.5 dropped the
# transitive torch dep). No index here: PyPI gives a CUDA-bundled wheel on
# win/linux x86_64 that runs fine on a CPU box (it falls back to CPU). For a lean
# CPU-only build the installer pre-installs the cpu-index wheel BEFORE this file
# (install-torch.mjs, cpu profile → https://download.pytorch.org/whl/cpu); the
# ==2.8.0 pins below then stay satisfied.
torch==2.8.0
torchaudio==2.8.0
