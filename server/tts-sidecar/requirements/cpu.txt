# CPU overlay (no GPU). Vendor-neutral base + the torch engines on a CPU runtime.

# (or ACCELERATOR=cpu). Differs from nvidia-cuda.txt only in the ONNX runtime:

-r base.txt

# Coqui XTTS is opt-in on every profile — install it from the Model Manager.
# Qwen is GPU-only (needs an nvidia/amd overlay) — it is not installable on a CPU-only box.

# torch + torchaudio — EXPLICIT, matched, pinned at 2.11.0 (same CVE rationale
# as the nvidia overlay). KEEP torch here even though a Kokoro-only CPU box
# never imports it: the Model Manager opt-installs Coqui via
# `pip install coqui-tts -c base.txt` (PLAIN, no [cpu] extra), which pulls NO
# torch — so the CPU profile must pre-pin torch for opt-in Coqui to import.
# No index here: install-torch.mjs (cpu profile) installs from whl/cpu.
torch==2.11.0
torchaudio==2.11.0

kokoro-onnx>=0.4.0,<0.5.0
