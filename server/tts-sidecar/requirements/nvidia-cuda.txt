# Use the maintained Coqui fork (the original `TTS` package was discontinued
# in early 2024 and won't install on Python >=3.12). `coqui-tts` keeps the
# `from TTS.api import TTS` import surface and supports Python 3.10–3.13.
# NOTE: we deliberately DROP the `[codec]` extra. `[codec]` pulls torchcodec,
# which (a) only ships cores for FFmpeg 4–7 (fails against the FFmpeg 8 we ship
# on PATH) and (b) is only NEEDED on torch >=2.9, where torchaudio's in-core
# load/save symbols moved out. We pin torch to 2.8 (below 2.9), so `torchaudio`
# still has in-core audio I/O and torchcodec is unnecessary. (Same approach as
# the AMD profile.)
coqui-tts>=0.24.0

# torch + torchaudio — EXPLICIT, matched, pinned. torch used to arrive
# transitively via coqui-tts, but coqui-tts 0.27.5 dropped that declaration, so a
# fresh venv had NO torch and Coqui XTTS + Qwen synth (≈10 `import torch` sites in
# main.py) failed (Kokoro = onnxruntime, unaffected).
#   - Pinned to the 2.8.0 matched pair: torchaudio is tightly coupled to torch's
#     exact version, and 2.8.x is the newest line where (a) a matching torchaudio
#     exists AND (b) torch < 2.9 keeps audio I/O in-core (no torchcodec needed).
#   - No index here (keeps macOS/CPU installs working — PyPI gives the cpu/mps
#     build there and a CUDA-bundled wheel on Windows/Linux x86_64). For a SPECIFIC
#     CUDA build, pre-install before this file, e.g. CUDA 12.8:
#       pip install torch==2.8.0 torchaudio==2.8.0 --index-url https://download.pytorch.org/whl/cu128
#     (the ==2.8.0 pins below then stay satisfied). See server/tts-sidecar/README.md.
torch==2.8.0
torchaudio==2.8.0

# NOTE on DeepSpeed: not in requirements because its setup.py needs the
# standalone NVIDIA CUDA Toolkit (with nvcc) at install time — the CUDA
# runtime libraries that ship inside the PyTorch wheel aren't enough. On
# Windows boxes without the toolkit, `pip install deepspeed` fails at
# metadata-generation. If you have (or are willing to install) the CUDA
# Toolkit from https://developer.nvidia.com/cuda-downloads, then:
#   DS_BUILD_OPS=0 ./.venv/Scripts/python.exe -m pip install --no-build-isolation "deepspeed>=0.15,<0.16"
# and set COQUI_DEEPSPEED=1 in server/.env. The sidecar already handles
# both states — it logs a warning and continues in vanilla mode if
# DeepSpeed isn't available.

# Kokoro v1 — quality-tuned local TTS engine, eagerly loaded at sidecar startup.
# Installed PLAIN (no [gpu] extra) so it deterministically lands the core
# `onnxruntime` (CPU) module; install-ort.mjs then swaps in the GPU runtime the
# profile needs (nvidia → onnxruntime-gpu) as the single enforcement point. We do
# NOT use kokoro-onnx[gpu]: that extra coexists with the core onnxruntime dep, and
# pip's resolution order can leave the CPU build owning the shared `onnxruntime/`
# namespace → a silent CPU-only Kokoro on a GPU box (the 2026-06-16 regression).
# macOS/Apple-Silicon keeps the plain CPU/CoreML runtime (no swap for non-nvidia).
# Run scripts/install-kokoro.{ps1,sh} once to fetch the model + voices manifest
# into voices/kokoro/ (gitignored, ~330 MB).
kokoro-onnx>=0.4.0,<0.5.0
